local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local ReplicatedStorage = game:GetService("ReplicatedStorage")

local Player = Players.LocalPlayer
local Modules = ReplicatedStorage:WaitForChild("Modules")
local Net = Modules:WaitForChild("Net")
local RegisterAttack = Net:WaitForChild("RE/RegisterAttack")
local RegisterHit = Net:WaitForChild("RE/RegisterHit")

local CONFIG = {
    RANGE = 200,
    ATTACKS_PER_FRAME = 15,
    HITS_PER_FRAME = 15,
    FRUIT_SPAM = 15
}

local LastTool = nil

local function IsAlive(m)
    local h = m:FindFirstChild("Humanoid")
    return h and h.Health > 0
end

local function GetTargets()
    local char = Player.Character
    if not char then return {} end

    local hrp = char:FindFirstChild("HumanoidRootPart")
    if not hrp then return {} end

    local list = {}

    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= Player and p.Character and p.Character:FindFirstChild("HumanoidRootPart") and IsAlive(p.Character) then
            local dist = (hrp.Position - p.Character.HumanoidRootPart.Position).Magnitude
            if dist <= CONFIG.RANGE then
                table.insert(list, p.Character)
            end
        end
    end

    if #list == 0 and workspace:FindFirstChild("Enemies") then
        for _, v in ipairs(workspace.Enemies:GetChildren()) do
            if v:FindFirstChild("HumanoidRootPart") and IsAlive(v) then
                local dist = (hrp.Position - v.HumanoidRootPart.Position).Magnitude
                if dist <= CONFIG.RANGE then
                    table.insert(list, v)
                end
            end
        end
    end

    return list
end

local function AttackMeleeSword(targets)
    for _ = 1, CONFIG.ATTACKS_PER_FRAME do
        RegisterAttack:FireServer(0)
    end

    local args = {[1]=targets[1].HumanoidRootPart,[2]={}}
    for i,v in ipairs(targets) do
        args[2][i] = {v, v.HumanoidRootPart}
    end

    for _ = 1, CONFIG.HITS_PER_FRAME do
        RegisterHit:FireServer(unpack(args))
    end
end

local function AttackFruit(targets, hrp)
    local char = Player.Character
    if not char then return end

    local tool = char:FindFirstChildOfClass("Tool")
    if not tool or tool.ToolTip ~= "Blox Fruit" then return end

    local remote = tool:FindFirstChild("LeftClickRemote")
    if not remote or not targets[1] then return end

    local dir = (targets[1].HumanoidRootPart.Position - hrp.Position).Unit

    for _ = 1, CONFIG.FRUIT_SPAM do
        remote:FireServer(dir, 1)
    end
end

RunService.Heartbeat:Connect(function()
    local char = Player.Character
    if not char then return end
    
    local hrp = char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end

    local tool = char:FindFirstChildOfClass("Tool")
    if tool ~= LastTool then
        LastTool = tool
    end

    local targets = GetTargets()
    if #targets == 0 then return end

    if tool and (tool.ToolTip == "Melee" or tool.ToolTip == "Sword") then
        AttackMeleeSword(targets)
    elseif tool and tool.ToolTip == "Blox Fruit" then
        AttackFruit(targets, hrp)
    end
end)
